import re

1、匹配某个字符串 

text = 'hello'
ret = re.match('he', text)
ret.group()
'he'

2、点 . 匹配任意字符 不能是\n 

text = 'ahello'
ret = re.match('.', text)
ret.group()
'a'

不能匹配换行符

text = '\n'
ret = re.match('.', text)
print(ret)
None

3、\d 匹配任意的数字 

text = '123456789'
ret = re.match('\d', text)
ret.group()
'1'
text = '0123456789'
ret = re.match('\d', text)
ret.group()
'0'

7、\W 与 \w 刚好相反 

text = '+'
ret = re.match('\W', text)
ret.group()
'+'

8、[],组合的方式只要满足中括号的内容，就可以匹配 

text = '0377-888888abcd888'
# + 代表一个或者多个满足条件
ret = re.match('[\d\-]+', text)
ret.group()
'0377-888888'

8.1 中括号形式 代替 \d

text = '123abc456'
ret = re.match('[0-9]+', text)
ret.group()
'123'
text = 'abc123abc456'
# ^ 取非
ret = re.match('[^0-9]+', text)
ret.group()
'abc'
